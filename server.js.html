/**

Initializes an instance of the Express application.
@const {function} express
*/
const express = require('express');
/**

Initializes an instance of the PostgreSQL connection pool.
@const {object} Pool
*/
const { Pool } = require('pg');
/**

Initializes an instance of the Express application.
@const {function} app
*/
const app = express();
/**

Initializes an instance of the PostgreSQL connection pool.
@const {object} pool
@param {object} connectionString - The connection string for the PostgreSQL database.
@param {string} user - The username for the PostgreSQL database.
@param {string} password - The password for the PostgreSQL database.
*/
const pool = new Pool({
connectionString: "jdbc:postgresql://csce-315-db.engr.tamu.edu/csce315331_team_62",
user: "csce315331_team_62_master",
password: "panda"
});
/**

Handles a GET request for the '/api/data' endpoint.
@function
@async
@param {object} req - The request object.
@param {object} res - The response object.
@throws {error} Throws an error if there is an issue querying the database.
*/
app.get('/api/data', async (req, res) => {
try {
const data = await pool.query('SELECT * FROM Inventory');
res.json(data.rows);
} catch (err) {
console.error(err);
res.sendStatus(500);
}
});
/**

Starts the Express server and listens for incoming connections on port 5000.
@function
@param {function} callback - The function to be called when the server starts listening.
*/
app.listen(5000, () => {
console.log('Server listening on port 5000');
});
