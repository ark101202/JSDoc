/**

Importing React and its Component class as well as the render function from react-dom.
*/
import React, { Component } from 'react';
import { render } from 'react-dom';
/**

Importing the BrowserRouter component from react-router-dom.
*/
import { BrowserRouter as Router } from 'react-router-dom';
/**

Importing the App component and the CSS file for the application.
*/
import App from './App';
import './style.css';
/**

Render the App component inside the Router component to enable client-side routing.
@function
@name render
@param {JSX.Element} <Router> - The top-level Router component for the application.
@param {JSX.Element} <App> - The main App component to render.
@param {HTMLElement} document.getElementById('root') - The DOM element where the App component should be rendered.
*/
render(
  <Router>
 <App />
</Router>,
document.getElementById('root')
);
