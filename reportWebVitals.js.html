/**
 * Calls web-vitals functions to measure key performance metrics of a web page and reports them via the provided callback function.
 *
 * @param {Function} onPerfEntry - The callback function to be called with the performance metrics. It takes a PerformanceEntry object as its parameter.
 * @returns {void}
 */

const reportWebVitals = onPerfEntry => {
  if (onPerfEntry && onPerfEntry instanceof Function) {
    import('web-vitals').then(({ getCLS, getFID, getFCP, getLCP, getTTFB }) => {
      getCLS(onPerfEntry);
      getFID(onPerfEntry);
      getFCP(onPerfEntry);
      getLCP(onPerfEntry);
      getTTFB(onPerfEntry);
    });
  }
};

export default reportWebVitals;
